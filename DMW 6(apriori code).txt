# Simple Apriori Implementation

from itertools import combinations

# Sample dataset
transactions = [
    ['milk', 'bread', 'butter'],
    ['bread', 'butter'],
    ['milk', 'bread'],
    ['milk', 'butter'],
    ['bread', 'butter']
]

min_support = 2

# Step 1: Get all unique items
items = set()
for transaction in transactions:
    for item in transaction:
        items.add(item)

# Step 2: Generate frequent 1-itemsets
freq_items = {}
for item in items:
    count = sum([1 for t in transactions if item in t])
    if count >= min_support:
        freq_items[(item,)] = count

print("Frequent 1-itemsets:", freq_items)

# Step 3: Generate 2-itemsets
pairs = list(combinations(items, 2))
freq_pairs = {}

for pair in pairs:
    count = sum([1 for t in transactions if pair[0] in t and pair[1] in t])
    if count >= min_support:
        freq_pairs[pair] = count

print("Frequent 2-itemsets:", freq_pairs)
























# Simple Apriori Implementation

# Import combinations function to generate pairs of items
from itertools import combinations


# -------------------------------
# STEP 1: DEFINE DATASET
# -------------------------------

# Each list inside 'transactions' represents one transaction
# Example: ['milk', 'bread'] means both items were bought together
transactions = [
    ['milk', 'bread', 'butter'],
    ['bread', 'butter'],
    ['milk', 'bread'],
    ['milk', 'butter'],
    ['bread', 'butter']
]

# Minimum support threshold
# Only itemsets appearing at least 2 times will be considered frequent
min_support = 2


# -------------------------------
# STEP 2: FIND UNIQUE ITEMS
# -------------------------------

# Create an empty set to store unique items
items = set()

# Loop through each transaction
for transaction in transactions:
    # Loop through each item in transaction
    for item in transaction:
        items.add(item)   # Add item to set (automatically avoids duplicates)


# -------------------------------
# STEP 3: FIND FREQUENT 1-ITEMSETS
# -------------------------------

# Dictionary to store item and its frequency
freq_items = {}

# Loop through each unique item
for item in items:
    # Count how many transactions contain this item
    count = sum([1 for t in transactions if item in t])

    # Check if it meets minimum support
    if count >= min_support:
        # Store item as a tuple (important format for Apriori)
        freq_items[(item,)] = count

# Print frequent single items
print("Frequent 1-itemsets:", freq_items)


# -------------------------------
# STEP 4: GENERATE 2-ITEMSETS
# -------------------------------

# Generate all possible pairs of items
pairs = list(combinations(items, 2))

# Dictionary to store frequent pairs
freq_pairs = {}

# Loop through each pair
for pair in pairs:
    # Count how many transactions contain BOTH items
    count = sum([1 for t in transactions if pair[0] in t and pair[1] in t])

    # Check minimum support
    if count >= min_support:
        freq_pairs[pair] = count

# Print frequent pairs
print("Frequent 2-itemsets:", freq_pairs)
